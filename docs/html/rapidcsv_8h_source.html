134 {

171 void ToVal(const std::string& pStr, T& pVal) const
172 {
173 try
174 {
175 if (typeid(T) == typeid(int))
176 {
177 pVal = static_cast<T>(std::stoi(pStr));
178 return;
179 }
180 else if (typeid(T) == typeid(long))
181 {
182 pVal = static_cast<T>(std::stol(pStr));
183 return;
184 }
185 else if (typeid(T) == typeid(long long))
186 {
187 pVal = static_cast<T>(std::stoll(pStr));
188 return;
189 }
190 else if (typeid(T) == typeid(unsigned))
191 {
192 pVal = static_cast<T>(std::stoul(pStr));
193 return;
194 }
195 else if (typeid(T) == typeid(unsigned long))
196 {
197 pVal = static_cast<T>(std::stoul(pStr));
198 return;
199 }
200 else if (typeid(T) == typeid(unsigned long long))
201 {
202 pVal = static_cast<T>(std::stoull(pStr));
203 return;
204 }
205 }
206 catch (...)
207 {
208 if (!mConverterParams.mHasDefaultConverter)
209 {
210 throw;
211 }
212 else
213 {
214 pVal = static_cast<T>(mConverterParams.mDefaultInteger);
215 return;
216 }
217 }
218
219 try
220 {
221 if (mConverterParams.mNumericLocale)
222 {
223 if (typeid(T) == typeid(float))
224 {
225 pVal = static_cast<T>(std::stof(pStr));
226 return;
227 }
228 else if (typeid(T) == typeid(double))
229 {
230 pVal = static_cast<T>(std::stod(pStr));
231 return;
232 }
233 else if (typeid(T) == typeid(long double))
234 {
235 pVal = static_cast<T>(std::stold(pStr));
236 return;
237 }
238 }
239 else
240 {
241 if ((typeid(T) == typeid(float)) ||
242 (typeid(T) == typeid(double)) ||
243 (typeid(T) == typeid(long double)))
244 {
245 std::istringstream iss(pStr);
246 iss.imbue(std::locale::classic());
247 iss >> pVal;
248 if (iss.fail() || iss.bad() || !iss.eof())
249 {
250 throw std::invalid_argument("istringstream: no conversion");
251 }
252 return;
253 }
254 }
255 }
256 catch (...)
257 {
258 if (!mConverterParams.mHasDefaultConverter)
259 {
260 throw;
261 }
262 else
263 {
264 pVal = static_cast<T>(mConverterParams.mDefaultFloat);
265 return;
266 }
267 }
268
269 if (typeid(T) == typeid(char))
270 {
271 pVal = static_cast<T>(pStr[0]);
272 return;
273 }
274 else
275 {
276 throw no_converter();
277 }
278 }

1490 void ReadCsv(std::istream& pStream)
1491 {
1492 Clear();
1493 pStream.seekg(0, std::ios::end);
1494 std::streamsize length = pStream.tellg();
1495 pStream.seekg(0, std::ios::beg);
1496
1497#ifdef HAS_CODECVT
1498 std::vector<char> bom2b(2, '\0');
1499 if (length >= 2)
1500 {
1501 pStream.read(bom2b.data(), 2);
1502 pStream.seekg(0, std::ios::beg);
1503 }
1504
1505 static const std::vector<char> bomU16le = { '\xff', '\xfe' };
1506 static const std::vector<char> bomU16be = { '\xfe', '\xff' };
1507 if ((bom2b == bomU16le) || (bom2b == bomU16be))
1508 {
1509 mIsUtf16 = true;
1510 mIsLE = (bom2b == bomU16le);
1511
1512 std::wifstream wstream;
1513 wstream.exceptions(std::wifstream::failbit | std::wifstream::badbit);
1514 wstream.open(mPath, std::ios::binary);
1515 if (mIsLE)
1516 {
1517 wstream.imbue(std::locale(wstream.getloc(),
1518 new std::codecvt_utf16<wchar_t, 0x10ffff,
1519 static_cast<std::codecvt_mode>(std::consume_header |
1520 std::little_endian)>));
1521 }
1522 else
1523 {
1524 wstream.imbue(std::locale(wstream.getloc(),
1525 new std::codecvt_utf16<wchar_t, 0x10ffff,
1526 std::consume_header>));
1527 }
1528 std::wstringstream wss;
1529 wss << wstream.rdbuf();
1530 std::string utf8 = ToString(wss.str());
1531 std::stringstream ss(utf8);
1532 ParseCsv(ss, static_cast<std::streamsize>(utf8.size()));
1533 }
1534 else
1535#endif
1536 {
1537 // check for UTF-8 Byte order mark and skip it when found
1538 if (length >= 3)
1539 {
1540 std::vector<char> bom3b(3, '\0');
1541 pStream.read(bom3b.data(), 3);
1542
1543 if (bom3b != s_Utf8BOM)
1544 {
1545 // file does not start with a UTF-8 Byte order mark
1546 pStream.seekg(0, std::ios::beg);
1547 }
1548 else
1549 {
1550 // file did start with a UTF-8 Byte order mark, simply skip it
1551 length -= 3;
1552 mHasUtf8BOM = true;
1553 }
1554 }
1555
1556 ParseCsv(pStream, length);
1557 }

1560 void ParseCsv(std::istream& pStream, std::streamsize p_FileLength)
1561 {
1562 const std::streamsize bufLength = 64 * 1024;
1563 std::vector<char> buffer(bufLength);
1564 std::vector<std::string> row;
1565 std::string cell;
1566 bool quoted = false;
1567 int cr = 0;
1568 int lf = 0;
1569
1570 while (p_FileLength > 0)
1571 {
1572 const std::streamsize toReadLength = std::min<std::streamsize>(p_FileLength, bufLength);
1573 pStream.read(buffer.data(), toReadLength);
1574
1575 // With user-specified istream opened in non-binary mode on windows, we may have a
1576 // data length mismatch, so ensure we don't parse outside actual data length read.
1577 const std::streamsize readLength = pStream.gcount();
1578 if (readLength <= 0)
1579 {
1580 break;
1581 }
1582
1583 for (size_t i = 0; i < static_cast<size_t>(readLength); ++i)
1584 {
1585 if (buffer[i] == mSeparatorParams.mQuoteChar)
1586 {
1587 if (cell.empty() || (cell[0] == mSeparatorParams.mQuoteChar))
1588 {
1589 quoted = !quoted;
1590 }
1591 else if (mSeparatorParams.mTrim)
1592 {
1593 // allow whitespace before first mQuoteChar
1594 const auto firstQuote = std::find(cell.begin(), cell.end(), mSeparatorParams.mQuoteChar);
1595 if (std::all_of(cell.begin(), firstQuote, [](int ch) { return isspace(ch); }))
1596 {
1597 quoted = !quoted;
1598 }
1599 }
1600 cell += buffer[i];
1601 }
1602 else if (buffer[i] == mSeparatorParams.mSeparator)
1603 {
1604 if (!quoted)
1605 {
1606 row.push_back(Unquote(Trim(cell)));
1607 cell.clear();
1608 }
1609 else
1610 {
1611 cell += buffer[i];
1612 }
1613 }
1614 else if (buffer[i] == '\r')
1615 {
1616 if (mSeparatorParams.mQuotedLinebreaks && quoted)
1617 {
1618 cell += buffer[i];
1619 }
1620 else
1621 {
1622 ++cr;
1623 }
1624 }
1625 else if (buffer[i] == '\n')
1626 {
1627 if (mSeparatorParams.mQuotedLinebreaks && quoted)
1628 {
1629 cell += buffer[i];
1630 }
1631 else
1632 {
1633 ++lf;
1634 if (mLineReaderParams.mSkipEmptyLines && row.empty() && cell.empty())
1635 {
1636 // skip empty line
1637 }
1638 else
1639 {
1640 row.push_back(Unquote(Trim(cell)));
1641
1642 if (mLineReaderParams.mSkipCommentLines && !row.at(0).empty() &&
1643 (row.at(0)[0] == mLineReaderParams.mCommentPrefix))
1644 {
1645 // skip comment line
1646 }
1647 else
1648 {
1649 mData.push_back(row);
1650 }
1651
1652 cell.clear();
1653 row.clear();
1654 quoted = false;
1655 }
1656 }
1657 }
1658 else
1659 {
1660 cell += buffer[i];
1661 }
1662 }
1663 p_FileLength -= readLength;
1664 }
1665
1666 // Handle last cell without linebreak
1667 if (!cell.empty())
1668 {
1669 row.push_back(Unquote(Trim(cell)));
1670 cell.clear();
1671 }
1672
1673 // Handle last line without linebreak
1674 if (!row.empty())
1675 {
1676 if (mLineReaderParams.mSkipCommentLines && !row.at(0).empty() &&
1677 (row.at(0)[0] == mLineReaderParams.mCommentPrefix))
1678 {
1679 // skip comment line
1680 }
1681 else
1682 {
1683 mData.push_back(row);
1684 }
1685
1686 row.clear();
1687 }
1688
1689 // Assume CR/LF if at least half the linebreaks have CR
1690 mSeparatorParams.mHasCR = (cr > (lf / 2));
1691
1692 // Set up column labels
1693 UpdateColumnNames();
1694
1695 // Set up row labels
1696 UpdateRowNames();
1697 }
